ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON; return ContentService.createTextOutput(output).setMimeType(mime); } function doGet(e) { const action = e.parameter.action; const callback = e.parameter.callback || ''; if (action === 'load') { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const result = {}; for (let i = 1; i < data.length; i++) { if (data[i][0]) { try { result[data[i][0]] = JSON.parse(data[i][1]); } catch(err) { result[data[i][0]] = data[i][1]; } } } return corsResponse({ ok: true, data: result }, callback); } if (action === 'saveAll') { try { const state = JSON.parse(decodeURIComponent(e.parameter.state || '{}')); const sheet = getSheet(); const lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.getRange(2,1,lastRow-1,2).clearContent(); const rows = Object.entries(state).map(([k,v]) => [k, JSON.stringify(v)]); if (rows.length > 0) sheet.getRange(2,1,rows.length,2).setValues(rows); return corsResponse({ ok: true }, callback); } catch(err) { return corsResponse({ ok: false, error: err.toString() }, callback); } } return corsResponse({ ok: false, error: 'Unknown action' }, callback); } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const { action, state } = body; if (action === 'saveAll') { const sheet = getSheet(); const lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.getRange(2,1,lastRow-1,2).clearContent(); const rows = Object.entries(state).map(([k,v]) => [k, JSON.stringify(v)]); if (rows.length > 0) sheet.getRange(2,1,rows.length,2).setValues(rows); return corsResponse({ ok: true }); } return corsResponse({ ok: false, error: 'Unknown action' }); } catch(err) { return corsResponse({ ok: false, error: err.toString() }); } }
